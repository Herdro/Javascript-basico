//Operadores matematicos (binarios)
a + b
a - b
a * b
a / b

//concatenar

//Negación
!false; //true

5 < 3; //false
5 > 3; //true
5 <= 6; //true

//Operador aditivo

a += b; //+b

//Comparar estricto (valor, tipo)
3 === "3"; //false
